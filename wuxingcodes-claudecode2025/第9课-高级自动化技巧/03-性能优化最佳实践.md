import asyncio
import statistics
from time import perf_counter

class BenchmarkRunner:
    def __init__(self):
        self.results = {}

    async def benchmark_function(self, func, *args, iterations=100):
        durations = []

        for _ in range(iterations):
            start = perf_counter()
            await func(*args)
            duration = perf_counter() - start
            durations.append(duration)

        result = {
            'mean': statistics.mean(durations),
            'median': statistics.median(durations),
            'stdev': statistics.stdev(durations),
            'min': min(durations),
            'max': max(durations)
        }

        self.results[func.__name__] = result
        return result

    def compare_implementations(self, old_func, new_func, *args):
        async def run_comparison():
            old_result = await self.benchmark_function(old_func, *args)
            new_result = await self.benchmark_function(new_func, *args)

            improvement = {
                'speedup': old_result['mean'] / new_result['mean'],
                'time_saved': old_result['mean'] - new_result['mean']
            }

            return {
                'old': old_result,
                'new': new_result,
                'improvement': improvement
            }

        return asyncio.run(run_comparison())
